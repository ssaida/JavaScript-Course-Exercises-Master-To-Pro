בתרגיל ג'אווה סקריפט הבא נלמד איך להשתמש בתנאי סוויץ switch, שיעזור לנו בחלק מהמקרים לקצר ולפשט תנאים מורכבים וארוכים לקוד שהרבה יותר קל לקרוא ולהבין אותו.

  • a. prompt the user and ask him what day is it today (as number from 1-7).
  • b. prompt how many days left until friday. "It's 3 Days Until Friday"
  • c. Bonus: if it's Thursday print "Friday is tomorrow!".
  • d. Bonus: if the number is not 1-7 print "is it maayan calendar day or what?". 

var day = +prompt('Which day is today?')
var daysUntilFriday
var dayName

switch (day) {
    case 1:
        dayName = 'Sunday'
        daysUntilFriday = 5
        break
    case 2:
        dayName = 'Monday'
        daysUntilFriday = 4
        break
    case 3:
        dayName = 'Tuesday'
        daysUntilFriday = 3
        break
    case 4:
        dayName = 'Wednesday'
        daysUntilFriday = 2
        break
    case 5:
        dayName = 'Thursday'
        daysUntilFriday = 1
        break
    case 6:
        dayName = 'Friday'
        daysUntilFriday = 0
        break
    case 7:
        dayName = 'Saturday'
        daysUntilFriday = 6
        break
    default:
        day = undefined
        dayName = 'invalid day'
}

if (daysUntilFriday === 0) {
    console.log('its Friday!')
} else if (daysUntilFriday === 1) {
    console.log('Friday is tomorrow!')
} else if (day) {
    console.log(
        'today is',
        dayName,
        'its ' + daysUntilFriday + ' Days until Friday.'
    )
} else {
    console.log('is it a maayan calendar or what?')
}
