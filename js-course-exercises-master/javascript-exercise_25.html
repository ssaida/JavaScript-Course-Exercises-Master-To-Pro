  • c. Find the index (position) of the searchTerm inside this string, e.g , searchTerm of character "n", in string "doctorcode" will result in 0 as the first char in the name doctorcode is “n” if we search for “l” the result is 6. 

const getIndexOf = (searchTerm, userInput) => {
    for (let i = 0; i < userInput.length; i++) {
        if (userInput[i] === searchTerm) {
            return i
        }
    }
    return false
}

const init = () => {
    const userInput = prompt('enter some word')
    const searchTerm = prompt('what char we are searching for?')
    const charIdx = getIndexOf(searchTerm, userInput)
    if (charIdx || charIdx === 0) {
        return console.log(searchTerm, 'found at index number:', charIdx)
    }
    return console.log(searchTerm, 'is cannot be found on', userInput)
}
