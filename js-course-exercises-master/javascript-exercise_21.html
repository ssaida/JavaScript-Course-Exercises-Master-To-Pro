לולאות תרגיל מספר 21 יצירת סיסמה רנדומלית

בתרגיל ג'אווה סקריפט הבא נלמד איך להשתמש בפונקציה שתייצר עבורנו 10 סיסמאות חזקות שעומדות בכל הקריטריונים של סיסמה חזקה ומאובטחת

  • a. prompt the user to enter the length of the password he likes to generate.
  • b. write a function generateRandomPassword() that generate a random password with the length the user asks.
  • c. you can use those characters 'ABCDEFGHIJKLMNOPabcdefghijklmnop123456789!@#$%^&*()'to pick up from.
  • d. print 10 different passwords like random password is: 7lYE#qrpXz. 

const generateRandomPassword = (passwordLength) => {
    let randomStr = ''
    const availableChar =
        'abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ01234567890!@#$%^&*()'
    for (let i = 0; i < passwordLength; i++) {
        const randomIdx = getRandomIntInclusive(1, availableChar.length - 1)
        randomStr += availableChar[randomIdx]
    }
    return randomStr
}

function getRandomIntInclusive(min, max) {
    min = Math.ceil(min)
    max = Math.floor(max)
    return Math.floor(Math.random() * (max - min + 1) + min)
}

const init = () => {
    const passwordLength = prompt('How long you want your password')
    if (!passwordLength) return
    for (let i = 0; i < 10; i++) {
        const randomPassword = generateRandomPassword(passwordLength)
        console.log('Random password is:', randomPassword)
    }
}
