לולאות תרגיל מספר 22 שינוי מחרוזת

בתרגיל הבא נלמד איך לעבוד עם לולאות ביחד עם מחרוזות ולעבור על כל אות במחרוזת ואף לבצע טרנספורמציה לאותיות מסוימות

  • a. prompt the user to enter his name.
  • b. change every second letter to an uppercase and alert the result. 

const init = () => {
    const name = prompt('what is your name?')
    let transformedName = ''
    if (!name) return
    for (let i = 0; i < name.length; i++) {
        if (i % 2 === 1) {
            transformedName += name[i].toUpperCase()
        } else {
            transformedName += name[i]
        }
    }
    alert(transformedName)
}
