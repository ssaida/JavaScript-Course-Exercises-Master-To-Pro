  • b. if numA is bigger > than numB print "1st number is bigger" else "2nd number is bigger".
  • c. Bonus: if numbers are equal === print "Numbers are equal". 

var firstNum = +prompt('enter 1st number')
var secNum = +prompt('enter 2nd number')

if (firstNum === secNum) {
    console.log('Numbers are equal!')
} else if (firstNum > secNum) {
    console.log('1st number is bigger')
} else {
    console.log('2nd number is bigger')
}
