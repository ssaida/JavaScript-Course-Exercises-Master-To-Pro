const printAsterisks = (numOfPrints) => {
    let currChar = '*'
    for (let i = 1; i < numOfPrints; i++) {
        currChar += '*'
    }
    console.log(currChar)
}

const init = () => {
    const triangleSize = +prompt('What is the triangle size you want?')
    for (let i = 1; i <= triangleSize; i++) {
        printAsterisks(i)
    }
    for (let i = triangleSize - 1; i > 0; i--) {
        printAsterisks(i)
    }
}
