בתרגיל ג'אווה סקריפט הבא נלמד לקבל ערך מהמשתמש ולבדוק באמצעות תנאי איזה סוג של ערך קיבלנו, האם זה ערך מסוג מחרוזת מספר שלם או אפילו ערך בוליאני

  • a. prompt the user to enter any val.
  • b. write a function typeOf() that expects this val and print its type.
  • c. for example sending a number of 10 will print "number"?, sending "123blabla"? will print "string"?
  • d. Bonus:find a way to support boolean too. 

const val = prompt('Enter some value!')
function typeOf(val) {
    if (val === 'false' || val === 'true') {
        console.log('valid as boolean')
    } else if (+val) {
        console.log('valid as number', +val)
    } else {
        console.log('valid as string', val)
    }
}

typeOf(val)
