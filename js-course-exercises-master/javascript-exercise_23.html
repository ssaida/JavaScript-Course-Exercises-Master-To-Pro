לולאות תרגיל מספר 23 מציאת פלינדרום

בתרגיל הבא נלמד איך באמצעות לולאה מוצאים פלינדרום, פלינדרום היא מילה שאם נהפוך אותה היא תקרא באותו הסדר של האותיות לדוגמה המילה אבא אמא שמש סוס וכו'

  • a. prompt the user to enter a string.
  • b. check if this string is a palindrome and print yes or no base on the results.
  • palindrome: a word, phrase, or sequence that reads the same backward as forward, e.g., madam | אבא 

const userInput = prompt('enter a palindrome').toLowerCase()
const palindrome = (input) => {
    const mid = input.length / 2
    for (let i = 0; i < mid; i++) {
        if (input[i] !== input[input.length - 1 - i]) {
            return false
        }
    }
    return true
}

const isPalindrome = palindrome(userInput)
console.log(
    userInput,
    isPalindrome ? 'is a Palindrome ' : 'is not a Palindrome'
)
