בתרגיל ג'אווה סקריפט הבא נלמד איך להשתמש בלולאת פור, נלמד את השימושים השונים של הלולאה ובתרגיל הבא נלמד אף איך אפשר להשתמש באינדקס הלולאה בשביל לחשב סכומים או ממוצעים

  • a. write a function sumNumbers() that prompt the user to enter how many iterations he would like to run, save it as numOfIterations(0-1000).
  • b. print all the numbers from 0 to numOfIterations number.
  • c. for example if the user enter 3:
    print the average of this iteration, "The average is: 2" .
    print the totalSum of this iteration, "The total sum is: 6" .
  • f. don't allow the function run if the user enter number above 1000 

const sumNumbers = () => {
    const numOfIterations = +prompt('How many iterations to run?')
    if (numOfIterations > 1000) {
        console.error('Too many iterations!')
        return
    }
    let sum = 0
    for (let i = 1; i <= numOfIterations; i++) {
        console.log(i)
        sum += i
    }
    console.log('The total sum is:', sum)
    console.log('The average is:', sum / numOfIterations)
}
