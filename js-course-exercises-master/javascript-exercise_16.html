ג'אווה סקריפט המדריך המלא 2023
לולאות תרגיל מספר 16 יצירת מספרי הרולטה

בתרגיל ג'אווה סקריפט הבא נלמד איך להשתמש בלולאות כדי לעבור על מספרים ספציפיים ולעשות עם המספרים הללו משהו, למשל להדפיס את כל מספרי הרולטה ובהתאם לכך את הצבעים של מספרי הרולטה

  • a. print all numbers of a roulette table.
  • b. print if the number is odd or even.
  • c. generate a lucky number from 0 - 36.
  • d. prompt the user to enter his lucky number, if the user lucky number is the same as the generated one print "You have won!" or "You have lost!". 

let rouletteMaxNumbers = 36

const printRouletteNumbers = () => {
    for (let i = 0; i <= rouletteMaxNumbers; i++) {
        console.log('Roulette number', i, 'is:', oddOrEven(i))
    }
}

const oddOrEven = (num) => (num % 2 ? 'odd' : 'even')

const init = () => {
    printRouletteNumbers()
    const luckyNumber = getRandomNumber(0, 36)
    const userLuckyNumber = +prompt('Enter your number!')
    console.log('User number was:', userLuckyNumber)
    console.log('lucky number:', luckyNumber)
    if (luckyNumber === userLuckyNumber) {
        console.log('You have won!')
    } else {
        console.log('You lost!')
    }
}

const getRandomNumber = (min, max) => {
    min = Math.ceil(min)
    max = Math.floor(max)
    return Math.floor(Math.random() * (max - min + 1) + min)
}
// Start!
init()
